$n = 100;   O(1)
$array[]= []; O(1)

for ($i = 0; $i < $n; $i++) {  O(n)
  for ($j = 1; $j < $n; $j *= 2) {  O(logn)
     $array[$i][$j]= true; O(1)
} }
